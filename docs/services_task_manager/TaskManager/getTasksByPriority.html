List<Task> getTasksByPriority()

Get tasks sorted by priority (highest first)

List<Task> getTasksByPriority() {
  final sortedTasks = List<Task>.from(_tasks);
  sortedTasks.sort((a, b) => b.priority.compareTo(a.priority));
  return sortedTasks;
}
